Exercise 1.8:
(define (cube x) (* x x x))

(define (good-enough? guess x)
  (< (abs (- (cube guess) x)) 0.00001))

(define (improve guess x)
   (/ (+ (/ x (* guess guess)) (* 2 guess)) 3))
  
(define (cbrt-iter guess x)
  (if (good-enough? guess x)
      guess
      (cbrt-iter (improve guess x)
                 x)))

(define (cbrt x)
  (cbrt-iter 1.0 x))
